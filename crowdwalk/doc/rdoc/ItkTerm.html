166 def toRuby(object, deeply = true)
167   if(!object.is_a?(Term)) then      # Term のインスタンスではない場合。
168     if(object.is_a?(Array) || object.is_a?(java.util.ArrayList)) then
169       return toRubyFromArray(object, deeply) ;
170     elsif(object.is_a?(Hash) || object.is_a?(java.util.HashMap)) then
171       return toRubyFromHash(object, deeply) ;
172     else
173       return object ;
174     end
175 
176   elsif(object.isNull()) then       # null Termの場合
177     return nil ;
178 
179   elsif(object.isAtom()) then               ## atom の場合
180     if(object.isInt()) then
181       return object.getInt() ;
182     elsif(object.isDouble()) then
183       return object.getDouble() ;
184     else
185       return object.getHead() ;
186     end
187 
188   elsif(object.isArray()) then              ## 配列の場合
189     return toRubyFromArray(object.getArray(), deeply) ;
190 
191   else                                      ## json の Object の場合
192     return toRubyFromHash(object.getBody(), deeply) ;
193   end
194 
195 end
